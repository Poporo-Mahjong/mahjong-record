<script>
  let results = JSON.parse(localStorage.getItem("results") || "[]");

  function addGame() {
    const r = Number(document.getElementById("rank").value);
    const s = Number(document.getElementById("score").value || 0);

    if (r < 1 || r > 4) return;

    results.push({ rank: r, score: s });
    localStorage.setItem("results", JSON.stringify(results));
    updateStats();
  }

  function updateStats() {
    const count = results.length;
    if (count === 0) return;

    const ranks = results.map(x => x.rank);
    const scores = results.map(x => x.score);

    const avg = ranks.reduce((a,b)=>a+b,0) / count;
    const avgScore = scores.reduce((a,b)=>a+b,0) / count;

    const top = ranks.filter(x=>x===1).length / count;
    const second = ranks.filter(x=>x===2).length / count;
    const third = ranks.filter(x=>x===3).length / count;
    const last = ranks.filter(x=>x===4).length / count;
    const ren = (ranks.filter(x=>x===1).length + ranks.filter(x=>x===2).length) / count;

    document.getElementById("count").textContent = "対局数: " + count;
    document.getElementById("avg").textContent = "平均着順: " + avg.toFixed(2);
    document.getElementById("avgScore").textContent = "平均点数: " + avgScore.toFixed(0);
    document.getElementById("top").textContent = "トップ率: " + (top*100).toFixed(1) + "%";
    document.getElementById("second").textContent = "2位率: " + (second*100).toFixed(1) + "%";
    document.getElementById("third").textContent = "3位率: " + (third*100).toFixed(1) + "%";
    document.getElementById("ren").textContent = "連帯率: " + (ren*100).toFixed(1) + "%";
    document.getElementById("last").textContent = "ラス率: " + (last*100).toFixed(1) + "%";
  }

  updateStats();
</script>
